function lnk {
    param(
        [string]$Url,
        [string]$NombreArchivo = "AccesoDirectoWeb"
    )
    $ShortcutPath = (Get-Location).Path + "\$NombreArchivo.lnk"
    $WScriptShell = New-Object -ComObject WScript.Shell
    $Shortcut = $WScriptShell.CreateShortcut($ShortcutPath)
    $Shortcut.TargetPath = $Url
    $Shortcut.Save()
    Write-Host "Acceso directo creado en: $ShortcutPath"
}

function lnk.sym{

    # Ruta de la carpeta actual
    $FolderPath = $PWD.Path

    # Obtener todos los archivos con la extensiÇÎÇZÇÎ¶ªn ".kicad_sym" en la carpeta actual y sus subcarpetas de forma recursiva
    $kicadSymFiles = Get-ChildItem -Path $FolderPath -Filter *.kicad_sym -Recurse

    # Crear accesos directos para cada archivo
    foreach ($file in $kicadSymFiles) {
        # Ruta completa del archivo de destino
        $targetFilePath = $file.FullName
        
        # Ruta para el archivo de acceso directo
        $shortcutFilePath = Join-Path -Path $FolderPath -ChildPath ($file.BaseName + '.lnk')

        # Crear el acceso directo
        $shell = New-Object -ComObject WScript.Shell
        $shortcut = $shell.CreateShortcut($shortcutFilePath)
        $shortcut.TargetPath = $targetFilePath
        $shortcut.Save()
    }

    # Obtener todos los accesos directos creados
    $shortcuts = Get-ChildItem -Path $FolderPath -Filter *.lnk -Recurse

    # Mover todos los accesos directos creados a la ubicaciÇÎÇZÇÎ¶ªn actual
    foreach ($shortcut in $shortcuts) {
        $shortcutFullName = $shortcut.FullName
        Move-Item -Path $shortcutFullName -Destination ".\" -Force
    }
}

# Variable global para almacenar la ÇÎÇZ¶ôÇæltima ubicaciÇÎÇZÇÎ¶ªn del archivo cortado (movido)
$global:LastCutItemLocation = $null
# FunciÇÎÇZÇÎ¶ªn para realizar un duplicado (con nombre modificado) de un archivo o carpeta en una ubicaciÇÎÇZÇÎ¶ªn especÇÎÇZÇÎÇ÷fica
function copy {
    param (
        [string]$NombreElemento,
        [string]$NuevoNombre = $null
    )

    if (-not $NombreElemento) {
        Write-Output "Por favor especifica el nombre del archivo o carpeta que deseas duplicar."
        return
    }

    if (-not $NuevoNombre) {
        # Obtener el nombre base y la extensiÇÎÇZÇÎ¶ªn del archivo o carpeta
        $NombreBase = [System.IO.Path]::GetFileNameWithoutExtension($NombreElemento)
        $Extension = [System.IO.Path]::GetExtension($NombreElemento)

        # Definir el nuevo nombre con el sufijo "1"
        $NuevoNombre = $NombreBase + "1" + $Extension
    }

    $Origen = Join-Path -Path $pwd.Path -ChildPath $NombreElemento

    if (Test-Path $Origen) {
        # Definir la ubicaciÇÎÇZÇÎ¶ªn de destino para el duplicado en D:\buffertemporal con el nuevo nombre
        $Destino = Join-Path -Path "D:\buffertemporal" -ChildPath $NuevoNombre

        # Copiar el archivo o carpeta a la ubicaciÇÎÇZÇÎ¶ªn de destino con el nuevo nombre
        Copy-Item -Path $Origen -Destination $Destino -Recurse
        Write-Output "Se ha realizado un duplicado de '$NombreElemento' como '$NuevoNombre' en la ubicaciÇÎÇZÇÎ¶ªn '$Destino'."
    } else {
        Write-Output "El archivo o carpeta '$NombreElemento' no existe en la ubicaciÇÎÇZÇÎ¶ªn actual."
    }
}



# FunciÇÎÇZÇÎ¶ªn para cortar (mover) un archivo desde la ubicaciÇÎÇZÇÎ¶ªn actual a una ubicaciÇÎÇZÇÎ¶ªn especÇÎÇZÇÎÇ÷fica
function cut {
    param (
        [string]$NombreElemento
    )

    if (-not $NombreElemento) {
        Write-Output "Por favor especifica el nombre del archivo o carpeta que deseas cortar (moviÇÎÇZ¶ôÇúndolo)."
        return
    }

    $Origen = Join-Path -Path $pwd.Path -ChildPath $NombreElemento

    if (Test-Path $Origen) {
        # Mover el archivo a la ubicaciÇÎÇZÇÎ¶ªn especÇÎÇZÇÎÇ÷fica para despuÇÎÇZ¶ôÇús pegarlo con paste
        $Destino = Join-Path -Path "D:\buffertemporal" -ChildPath $NombreElemento
        Move-Item -Path $Origen -Destination $Destino -Force
        $global:LastCutItemLocation = $Destino  # Almacenar la ubicaciÇÎÇZÇÎ¶ªn del archivo cortado
        Write-Output "Se ha cortado (movido) '$NombreElemento' a la ubicaciÇÎÇZÇÎ¶ªn '$Destino'."
    } else {
        Write-Output "El archivo o carpeta '$NombreElemento' no existe en la ubicaciÇÎÇZÇÎ¶ªn actual."
    }
}

# FunciÇÎÇZÇÎ¶ªn para pegar el archivo cortado desde una ubicaciÇÎÇZÇÎ¶ªn especÇÎÇZÇÎÇ÷fica a otra ubicaciÇÎÇZÇÎ¶ªn
function paste {
    param (
        [string]$Destino
    )

    if (-not $Destino) {
        Write-Output "Por favor especifica la ubicaciÇÎÇZÇÎ¶ªn de destino donde deseas pegar el archivo."
        return
    }

    if (-not $global:LastCutItemLocation) {
        Write-Output "No hay archivos cortados para pegar."
        return
    }

    # Obtener el nombre base del archivo cortado
    $NombreElemento = [System.IO.Path]::GetFileName($global:LastCutItemLocation)
    $DestinoCompleto = Join-Path -Path $Destino -ChildPath $NombreElemento

    # Pegar el archivo cortado en la ubicaciÇÎÇZÇÎ¶ªn de destino
    Move-Item -Path $global:LastCutItemLocation -Destination $DestinoCompleto -Force
    $global:LastCutItemLocation = $null  # Limpiar la variable despuÇÎÇZ¶ôÇús de pegar
    Write-Output "Se ha pegado el archivo '$NombreElemento' en la ubicaciÇÎÇZÇÎ¶ªn '$Destino'."
}




# FunciÇÎÇZÇÎ¶ªn para eliminar un archivo o carpeta
function delete {
    Param(
        [string]$Name
    )

    # Verificar si el elemento a eliminar es un archivo o una carpeta
    if (Test-Path $Name -PathType Leaf) {
        # Si es un archivo, eliminarlo
        Remove-Item -Path $Name -Force
    } elseif (Test-Path $Name -PathType Container) {
        # Si es una carpeta, eliminarla
        Remove-Item -Path $Name -Recurse -Force
    } else {
        Write-Host "El elemento especificado no existe."
    }
    ff
}

function dd ($number) {
    $Archivos = Get-ChildItem
    $directories = $Archivos | Where-Object { $_.PSIsContainer }  # Filtra solo directorios

    if ($number -le $directories.Count -and $number -gt 0) {
        $directory = $directories[$number - 1].FullName
        Set-Location $directory
        Write-Output "Moved to: $directory"
    } else {
        Write-Output "Invalid number: The number must be within the range of listed directories."
    }
    ff
}

function Encrypt-File {
    param (
        [string]$FilePath,
        [string]$Password
    )

    $plainText = Get-Content $FilePath -Raw
    $plainBytes = [System.Text.Encoding]::UTF8.GetBytes($plainText)

    $sha256 = [System.Security.Cryptography.SHA256]::Create()
    $key = $sha256.ComputeHash([System.Text.Encoding]::UTF8.GetBytes($Password))

    $aes = [System.Security.Cryptography.Aes]::Create()
    $aes.Key = $key
    $aes.GenerateIV()

    $encryptor = $aes.CreateEncryptor($aes.Key, $aes.IV)
    $ms = New-Object System.IO.MemoryStream
    $cs = New-Object System.Security.Cryptography.CryptoStream $ms, $encryptor, [System.Security.Cryptography.CryptoStreamMode]::Write
    $cs.Write($plainBytes, 0, $plainBytes.Length)
    $cs.Close()

    $encryptedBytes = $ms.ToArray()
    $result = [byte[]]($aes.IV + $encryptedBytes)
    [System.IO.File]::WriteAllBytes($FilePath, $result)

    Write-Host "File encrypted successfully."
}

function Decrypt-File {
    param (
        [string]$FilePath,

    )

    $encryptedBytes = [System.IO.File]::ReadAllBytes($FilePath)
    $aes = [System.Security.Cryptography.Aes]::Create()

    $sha256 = [System.Security.Cryptography.SHA256]::Create()
    $key = $sha256.ComputeHash([System.Text.Encoding]::UTF8.GetBytes($Password))

    $iv = $encryptedBytes[0..15]
    $cipherText = $encryptedBytes[16..($encryptedBytes.Length - 1)]

    $aes.Key = $key
    $aes.IV = $iv
    $decryptor = $aes.CreateDecryptor($aes.Key, $aes.IV)

    $ms = New-Object System.IO.MemoryStream
    $cs = New-Object System.Security.Cryptography.CryptoStream $ms, $decryptor, [System.Security.Cryptography.CryptoStreamMode]::Write
    $cs.Write($cipherText, 0, $cipherText.Length)
    $cs.Close()

    $plainBytes = $ms.ToArray()
    $plainText = [System.Text.Encoding]::UTF8.GetString($plainBytes)
    
    Write-Output $plainText
}

function qqcatsafe1 {
    param (
        [int]$number
    )
    
    $password = Read-Host "Enter password" -AsSecureString
    $plainPassword = [Runtime.InteropServices.Marshal]::PtrToStringAuto([Runtime.InteropServices.Marshal]::SecureStringToBSTR($password))

    $Items = Get-ChildItem
    if ($number -le $Items.Count -and $number -gt 0) {
        $item = $Items[$number - 1]
        if (-not $item.PSIsContainer) {
            $path = $item.FullName
            
            Write-Host "Archivo: $($item.Name)" -BackgroundColor Yellow
            Write-Host ''
            
            try {
                $content = Decrypt-File -FilePath $path -Password $plainPassword
                Write-Output $content
            } catch {
                Write-Host "Cannot open the file: $path. There may be an issue with file association or access permissions."
                Write-Host "Error details: $_"
            }
        } else {
            Write-Host "The selected item is a directory and cannot be opened."
        }
    } else {
        Write-Host "Invalid number: The number must be within the range of listed files."
    }
    Write-Host ''
}
